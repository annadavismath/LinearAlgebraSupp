\begin{exercise}
Suppose $\vec{v}_1, \vec{v}_2, \vec{v}_3$ and $\vec{w}$ are vectors in $\RR^n$.  What are the implications of the statement: \textit{$\vec{w}$ is in the span of $\vec{v}_1, \vec{v}_2, \vec{v}_3$}
Select all that apply.
\begin{selectAll}
 \choice[correct]{Vector $\vec{w}$ a linear combination of $\vec{v}_1, \vec{v}_2, \vec{v}_3$}
 \choice[correct]{The set $\{\vec{v}_1, \vec{v}_2, \vec{v}_3, \vec{w}\}$ contains redundant vectors.}
 \choice{Vectors $\vec{v}_1, \vec{v}_2, \vec{v}_3, \vec{w}$ are linearly independent.}
 \choice[correct]{Vectors $\vec{v}_1, \vec{v}_2, \vec{v}_3, \vec{w}$ are linearly dependent.}
 \choice{Vector $\vec{v}_1$ can be written as a linear combination of $\vec{v}_2, \vec{v}_3, \vec{w}$.}
 \choice[correct]{The equation $a\vec{v}_1+ b\vec{v}_2+ c\vec{v}_3+ d\vec{w}=\vec{0}$ has a non-trivial solution.}
 \end{selectAll}
\end{exercise}
