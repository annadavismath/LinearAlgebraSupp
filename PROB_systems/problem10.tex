\begin{exercise}
Consider the following augmented matrix in which $\ast$ denotes an
arbitrary number and $\blacksquare$ denotes a nonzero number. 

\begin{equation*}
\left[
\begin{array}{ccccc|c}
\blacksquare & \ast & \ast & \ast & \ast & \ast \\
0 & \blacksquare & \ast & \ast & \ast & \ast \\
0 & 0 & \blacksquare & \blacksquare & \ast & \ast \\
0 & 0 & 0 & 0 & \blacksquare & 0
\end{array}
\right]
\end{equation*}
 
Which of the following is true?  Select all that apply.

\begin{selectAll}
 \choice[correct]{The corresponding system of equations is consistent.}
  \choice{The corresponding system of equations is inconsistent.}
 \choice[correct]{The rank of the augmented matrix is $4$.}
 \choice{The rank of the augmented matrix is $5$.}
 \choice{The rank of the augmented matrix is $6$.}
  \choice{There are no free variables.}
 \choice[correct]{There is one free variable.}
 \choice{There are two free variables.}
 \choice{The corresponding system has a unique solution.}
 \choice[correct]{The corresponding system has infinitely many solutions.}
 \end{selectAll}

\end{exercise}
