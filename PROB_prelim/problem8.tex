\begin{exercise}
Find $\vec{v}\times \vec{w}$ if 
$$\vec{v}=\begin{bmatrix}2\\-1\\4\end{bmatrix};\quad \vec{w}=\begin{bmatrix}3\\-3\\1\end{bmatrix}$$
$$\vec{v}\times\vec{w}=\begin{bmatrix}\answer{11}\\\answer{10}\\\answer{-3}\end{bmatrix}$$

Which of the following is true about $\vec{v}\times \vec{w}$?
\begin{multipleChoice}
  \choice{$\norm{\vec{v}\times \vec{w}}$ is the product of $\norm{\vec{v}}$ and $\norm{\vec{w}}$.}
 \choice{$\vec{v}\times \vec{w}$ is parallel to ONE of the original vectors.}
 \choice{$\vec{v}\times \vec{w}$ is orthogonal to ONE of the original vectors.}
 \choice[correct]{$\vec{v}\times \vec{w}$ is orthogonal to BOTH of the original vectors.}
 \end{multipleChoice}
\end{exercise}
