\begin{exercise}
If $A = \begin{bmatrix}
5 & 0 & 0 \\
0 & 3 & 7 \\
0 & 7 & 3
\end{bmatrix}$, find an orthogonal matrix $Q$ such that $Q^{-1}AQ$ is diagonal.

$Q = \begin{bmatrix}
1 & 0 & 0 \\
\answer{0} & 1/\sqrt{2} & 1/\sqrt{2} \\
\answer{0} & 1/\sqrt{2} & \answer{-1/\sqrt{2}}
\end{bmatrix}$

 \end{exercise}
