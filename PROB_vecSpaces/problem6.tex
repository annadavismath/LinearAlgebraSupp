\begin{exercise}
Let $V$ denote the set of ordered triples $(x, y, z)$ and define addition in $V$ as in $\RR^3$.  For each of the following definitions of scalar multiplication, select all properties of scalar multiplication for vector spaces that hold and decide whether $V$ is a vector space.

\begin{enumerate}
\item
    $$a(x,y,z)=(ax, y, az)$$
Select all properties of scalar multiplication that hold true.
\begin{selectAll}
\choice[correct]{Closure under scalar multiplication}
    \choice[correct]{Distributive Property over Vector Addition:\quad
  $k(\vec{u}+\vec{v})=k\vec{u}+k\vec{v}$}
  \choice[correct]{Distributive Property over Scalar Addition:\quad
  $(k+p)\vec{u}=k\vec{u}+p\vec{u}$}
  \choice[correct]{Associative Property for Scalar Multiplication:\quad
  $k(p\vec{u})=(kp)\vec{u}$}
  \choice[correct]{Multiplication by $1$:\quad
  $1\vec{u}=\vec{u}$}
\end{selectAll}

Is $V$ a vector space?
\begin{multipleChoice}
 \choice[correct]{$V$ is a vector space.}
 \choice{$V$ is not a vector space.}
 \end{multipleChoice}
